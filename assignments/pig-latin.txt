    function translatePig(){

        //gets original text
        var originalText = $("#text_input").val()

        //splits the words into array and removes most symbols
        var splitText = originalText.split(/[\s,.!?;-]+/)
        
        //removes any empty array elements
        splitText.forEach(function(el,index){
            if (el == ""){
                splitText.pop(index)
            }
        });

        //translates into pig latin
        var translatedSentance = splitText.map(function(word){

            // makes each word into an array
            word = word.split("")

            //moves the first letter and ads 'ay'     
            word.push(word.shift() + "ay")

            //makes into a string again
            word = word.join("")
            return word;
        })

        //joins the array so it display as 1 sentance without comas ,
        translatedSentance = translatedSentance.join(" ")

        //displays the text to the user
        $("#translated-text").text(translatedSentance)
    }
